1 primes :: [Int]
 2 primes = 2 : primes'
 3   where isPrime (p:ps) n = p*p > n ∨ n `rem` p ≢ 0 ∧ isPrime ps n
 4         primes' = 3 : filter (isPrime primes') [5, 7 …]
 5 
 6 primeLim :: [Int]
 7 primeLim = takeWhile (λx →  x < 100000000) primes
 8 
 9 genAll :: Int →  [[Int]]
10 genAll n = map (λx →  n:[x]) (takeWhile (λx →  x ≤ (100000000 ÷ n)) primeLim)
11 
12 primeTable :: Int →  [Int]
13 primeTable x = [if (i `mod` x ≡ 0) then x else 0 | i ←  [1…100000000]]
14 
15 allPrimeTable :: [[Int]]
16 allPrimeTable = [primeTable x | x ←  primeLim]
17 
18 primeFactorTable :: [[Int]]
19 primeFactorTable = [filter (λx →  x ≢ 0) [(allPrimeTable ‼ y) ‼ x | y ←  [0…(length allPrimeTable - 1)]] | x ←  [0…99999999]]

21 totient :: Int →  [Int] →  Int
22 totient x y = numerator ÷ denominator
23   where numerator = x * (∏(map (λx →  x - 1) y))
24         denominator = ∏y
